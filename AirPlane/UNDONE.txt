bool bullet_collide(float& xBullet , float& yBullet,
             float& xCreep, float& yCreep,
             float& xBoss , float& yBoss,
             float& xCreep1, float& yCreep1,
             float& xBoss1, float& yBoss1) {
    if ( (xBullet <= xCreep+30 && yBullet <= yCreep+40 && xBullet >= xCreep-42 && yBullet >= yCreep-44)||
         (xBullet <= xBoss+60 && yBullet <= yBoss+85 && xBullet >= xBoss-45 && yBullet >= yBoss-45) ||
         (xBullet <= xCreep1+30 && yBullet <= yCreep1+40 && xBullet >= xCreep1-42 && yBullet >= yCreep1-44) ||
         (xBullet <= xBoss1+60 && yBullet <= yBoss1+85 && xBullet >= xBoss1-45 && yBullet >= yBoss1-45) ) {
        return true;
    }
    return false;
}
